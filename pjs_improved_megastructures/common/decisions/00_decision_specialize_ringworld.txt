decision_ringworld_specialization_arms_industry = {
	owned_planets_only = yes

	prerequisites = {
		"tech_ring_world_construction"
	}

	enactment_time = 0
	resources = {
		category = decisions
		cost = {
			influence = 25
		}
	}

	potential = {
        is_planet_class = pc_ringworld_habitable
        AND = {
            NOT = { has_modifier = ringworld_specialization_arms_industry }
            NOT = { has_modifier = ringworld_specialization_civilian_industry }
            NOT = { has_modifier = ringworld_specialization_leisure }
            NOT = { has_modifier = ringworld_specialization_science }
            NOT = { has_modifier = ringworld_specialization_refinery }
        }
	}

    effect = {
	    remove_modifier = ringworld_specialization_none
        add_modifier = { modifier = ringworld_specialization_arms_industry days = -1 }
	}

	ai_weight = {
		weight = 1
	}
}

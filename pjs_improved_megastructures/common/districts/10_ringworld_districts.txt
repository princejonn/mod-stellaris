		}
		modifier = {

		}
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_maintenance_drone_effect_desc
	}

	ai_weight = {
		weight = 10
		modifier = {
			weight = 90000
			planet = {
				free_housing < 10
				free_amenities < 10
			}
			owner = {
				has_resource = {
					type = minerals
					amount > 1000
				}
			}
		}
		modifier = {
			weight = 90000
			planet = {
				num_districts = { type = district_ringworld_city value < 2 }
			}
		}
		modifier = {
			weight = 90000
			planet = {
				owner = { has_ascension_perk = ap_master_builders }

district_ringworld_arms_industry = {
	base_buildtime = 800
	is_capped_by_modifier = no

	show_on_uncolonized = {
	    has_modifier = ringworld_specialization_arms_industry
	}

	potential = {
	    exists = owner
	    has_modifier = ringworld_specialization_arms_industry
	}

	resources = {
		category = planet_districts
		cost = {
		    minerals = 1600
		}
		upkeep = {
		    energy = 10
		}
	}

	planet_modifier = {
		planet_housing_add = 20
	}


	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		modifier = {
			job_foundry_add = 20
		}
	}
	triggered_desc = {
		trigger = {
			exists = owner

		potential = {
			exists = owner

		}

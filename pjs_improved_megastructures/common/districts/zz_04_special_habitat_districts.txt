district_hab_science = {
	base_buildtime = 480
	is_capped_by_modifier = no

	show_on_uncolonized = {
		is_planet_class = pc_habitat
	    NOT = { has_modifier = habitat_specialization_astro_mining }
	    NOT = { has_modifier = habitat_specialization_hydroponics }
	    NOT = { has_modifier = habitat_specialization_exotic_gases }
	    NOT = { has_modifier = habitat_specialization_rare_crystals }
	    NOT = { has_modifier = habitat_specialization_volatile_motes }
	}

	potential = {
		exists = owner
		is_planet_class = pc_habitat
	    NOT = { has_modifier = habitat_specialization_astro_mining }
	    NOT = { has_modifier = habitat_specialization_hydroponics }
	    NOT = { has_modifier = habitat_specialization_exotic_gases }
	    NOT = { has_modifier = habitat_specialization_rare_crystals }
	    NOT = { has_modifier = habitat_specialization_volatile_motes }
	}

	resources = {
		category = planet_districts
		cost = {
			minerals = 600
		}
		upkeep = {
			energy = 5
		}
	}

	planet_modifier = {
		planet_housing_add = 8
	}


	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		modifier = {
			job_researcher_add = 8
		}
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		text = job_researcher_effect_desc
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		modifier = {
			job_brain_drone_add = 8
		}
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		text = job_brain_drone_effect_desc
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		modifier = {
			job_calculator_add = 8
		}
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		text = job_calculator_effect_desc
	}


	ai_weight = {
		weight = 10
		modifier = {
			weight = 90000
            owner = {
                has_monthly_income = {
                    resource = consumer_goods
                    value > 10
                }
            }
			planet = {
				num_districts = { type = district_hab_science value < 2 }
			}
		}
	}
}

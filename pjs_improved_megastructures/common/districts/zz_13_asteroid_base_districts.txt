	base_buildtime = 360
	is_capped_by_modifier = no

	    exists = owner
	    has_modifier = asteroid_base
	}

		NOT = { has_modifier = resort_colony }

			free_jobs < 2
			controller = { is_ai = no }
			owner = { is_ai = no }
			gai_good_minerals = yes
			free_district_slots = 0
		}

		category = planet_districts

			energy = 1
		}
	}

		potential = {
			exists = owner

				is_gestalt = no
				is_fallen_empire_spiritualist = no
			}
		}
		modifier = {
			job_miner_add = 4
		}
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				is_gestalt = no

			}
		}
		text = job_miner_effect_desc
	}


	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		modifier = {
			job_mining_drone_add = 4
		}
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		text = job_mining_drone_effect_desc
	}


	triggered_planet_modifier = {
		potential = {
			exists = owner

		}
		modifier = {
			job_mining_drone_add = 4
		}
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		text = job_mining_drone_effect_desc
	}


	triggered_planet_modifier = {

			job_fe_acolyte_mine_add = 4

		trigger = {

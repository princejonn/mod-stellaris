district_ringworld_rural = {
	base_buildtime = 240
	is_capped_by_modifier = no

	show_on_uncolonized = {
		is_planet_class = pc_ringworld_habitable
	}

	potential = {
		is_planet_class = pc_ringworld_habitable
	}

	resources = {
		category = planet_districts
		cost = {
			minerals = 500
		}
		upkeep = {
			energy = 2
		}
	}

	planet_modifier = {
		planet_housing_add = 2
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_valid_civic = civic_agrarian_idyll }
		}
		modifier = {
			planet_housing_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_agrarian_idyll
				has_valid_civic = civic_agrarian_idyll
			}
		}
		modifier = {
			planet_housing_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		modifier = {
			job_technician_drone_add = 1
			job_agri_drone_add = 1
		}
	}
    triggered_desc = {
        trigger = {
			exists = owner
			owner = { is_hive_empire = yes }
        }
        text = job_district_ringworld_rural_hive_desc
    }

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
			    is_machine_empire = yes
                has_valid_civic = civic_machine_servitor
            }
		}
		modifier = {
			job_technician_drone_add = 1
			job_agri_drone_add = 1
		}
	}
    triggered_desc = {
        trigger = {
			exists = owner
			owner = {
			    is_machine_empire = yes
                has_valid_civic = civic_machine_servitor
            }
        }
        text = job_district_ringworld_rural_servitor_desc
    }

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
			    is_machine_empire = yes
                NOT = { has_valid_civic = civic_machine_servitor }
            }
		}
		modifier = {
			job_technician_drone_add = 2
		}
	}
    triggered_desc = {
        trigger = {
			exists = owner
			owner = {
			    is_machine_empire = yes
                NOT = { has_valid_civic = civic_machine_servitor }
            }
        }
        text = job_district_ringworld_rural_machine_desc
    }

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_gestalt = no
				is_fallen_empire_spiritualist = no
			}
		}
		modifier = {
			job_technician_add = 1
			job_farmer_add = 1
		}
	}
    triggered_desc = {
        trigger = {
			exists = owner
			owner = {
				is_gestalt = no
				is_fallen_empire_spiritualist = no
			}
        }
        text = job_district_ringworld_rural_regular_desc
    }

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_fallen_empire_spiritualist = yes }
		}
		modifier = {
			job_fe_acolyte_generator_add = 1
			job_fe_acolyte_farm_add = 1
		}
	}
    triggered_desc = {
        trigger = {
			exists = owner
			owner = { is_fallen_empire_spiritualist = yes }
        }
        text = job_district_ringworld_rural_fe_spiritualist_desc
    }

	ai_weight = {
		weight = 2
	}
}

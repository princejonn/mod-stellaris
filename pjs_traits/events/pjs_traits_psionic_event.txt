namespace = pjs_traits_psionic_event

event = {
	id = pjs_traits_psionic_event.1

	hide_window = yes
	is_triggered_only = yes

	immediate = {
		every_country = {
			limit = { is_psionic_trait = yes }
			country_event = { id = pjs_traits_psionic_event.2 }
		}
	}
}

	id = pjs_traits_psionic_event.2

	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = { NOT = { has_technology = tech_psionic_theory } }
			add_research_option = tech_psionic_theory
		}
	}
}

building_crystal_plant = {
	base_buildtime = 720

	category = manufacturing

	potential = {
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
	}

	allow = {
		has_upgraded_capital = yes
	}

	destroy_trigger = {
		OR = {
			##gai_destroy_building = yes
			AND = {
				planet = {
					free_building_slots = 0
					exists = owner
					owner = {
						is_ai = yes
						has_monthly_income = {
							resource = rare_crystals
							value > 9
						}
					}
				}
			}
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			job_translucer_add = 2
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_translucer_drone_add = 2
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 720
		}
		upkeep = {
			energy = 7
		}
	}

	prerequisites = {
		"tech_rare_crystals"
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_translucer_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_translucer_drone_effect_desc
	}

	ai_weight = {
		weight = 0
		modifier = {
			weight = 30
			owner = {
				has_monthly_income = {
					resource = rare_crystals
					value < 3
				}#5 is too high since that means 2 extra buildings. Better to just go with 1 extra building
				NOT = {
					any_owned_planet = {
						has_building_construction = building_crystal_plant
					}
				}#Prevents mass spamming of building
			}
			gai_build_building = yes
		}
		modifier = {
			factor = 0
			OR = {
				gai_focus_world = yes
			}
		}
	}

	ai_resource_production = {
		rare_crystals = 1
		trigger = {
			always = yes
		}
	}
}

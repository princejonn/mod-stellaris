namespace = pjs_systems_sol_event

planet_event = {
	id = pjs_systems_sol_event.1
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = { has_planet_flag = planet_earth }

			## energy
			add_deposit = d_geothermal_vent
			add_deposit = d_frozen_gas_lake
			add_deposit = d_tempestous_mountain

			## minerals
			add_deposit = d_rich_mountain
			add_deposit = d_submerged_ore_veins
			add_deposit = d_prosperous_mesa

			## food
			add_deposit = d_black_soil
			add_deposit = d_tropical_island
			add_deposit = d_great_river
		}

		if = {
		    limit = {
		        has_planet_flag = sol_augmented_start
		        owner = { is_regular_empire = yes }
            }

            while = {
                count = 4
                add_district = district_city
            }
            while = {
                count = 3
                add_district = district_generator
            }
            while = {
                count = 3
                add_district = district_mining
            }
            while = {
                count = 3
                add_district = district_farming
            }

            add_building = building_factory_1
            add_building = building_holo_theatres
            add_building = building_commercial_zone
            add_building = building_research_lab_1

            while = {
                count = 33
                create_pop = { species = owner_main_species }
            }

            owner = {
                give_technology = { tech = "tech_physics_1" message = no }
                give_technology = { tech = "tech_physics_2" message = no }
                give_technology = { tech = "tech_society_1" message = no }
                give_technology = { tech = "tech_society_2" message = no }
                give_technology = { tech = "tech_engineering_1" message = no }
                give_technology = { tech = "tech_engineering_2" message = no }
                give_technology = { tech = "tech_automated_exploration" message = no }

                add_resource = { energy = 500 }
                add_resource = { minerals = 500 }
                if = {
                    limit = { is_machine_empire = no }
                    add_resource = { food = 500 }
                }
                add_resource = { alloys = 500 }

                if = {
                    limit = { is_gestalt = no }
                    add_resource = { consumer_goods = 200 }
                }
                add_resource = { influence = 200 }
                add_resource = { unity = 200 }
            }
		}
	}
}

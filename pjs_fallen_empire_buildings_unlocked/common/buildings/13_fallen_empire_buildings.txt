building_affluence_center = {
	base_buildtime = 900
	can_build = yes
	can_demolish = yes

	category = resource

	potential = {
        exists = owner
	    OR = {
            owner = { is_fallen_empire = yes }
	        owner = { has_technology = "tech_affluence_center" }
	    }
	}

	resources = {
		category = planet_buildings
		produces = {
			consumer_goods = 50
		}
		cost = {
			minerals = 1200
			rare_crystals = 600
		}
		upkeep = {
			minerals = 25
			energy = 10
		}
	}

	on_queued = {
		owner = {
			subtract_variable = { which = affluence_center_build_count value = 1 }
		}
	}

	on_unqueued = {
		owner = {
			change_variable = { which = affluence_center_build_count value = 1 }
		}
	}
}

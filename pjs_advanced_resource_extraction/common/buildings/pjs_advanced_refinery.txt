## overload
building_refinery = {
	base_buildtime = 480

	category = manufacturing

	potential = {
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
	}

	allow = {
		has_upgraded_capital = yes
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			job_gas_refiner_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_gas_refiner_drone_add = 1
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 500
		}
		upkeep = {
			energy = 3
		}
	}

	prerequisites = {
		"tech_exotic_gases"
	}

	upgrades = {
		building_refinery_2
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_gas_refiner_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_gas_refiner_drone_effect_desc
	}

	ai_weight = {
		weight = 2
	}

	ai_resource_production = {
		exotic_gases = 1
		trigger = {
			always = yes
		}
	}
}

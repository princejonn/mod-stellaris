	hide_window = yes
	is_triggered_only = yes

	immediate = {
        if = {
            limit = {
                AND = {
                    OR = {
                        is_gestalt = no
                        has_valid_civic = civic_machine_servitor
                    }
                    NOT = { has_ethic = ethic_spiritualist }
                    NOT = { has_ethic = ethic_fanatic_spiritualist }
                }
            }

            pjs_give_soc_tech_nonspiritualist_or_servitor = yes
        }
    }
}

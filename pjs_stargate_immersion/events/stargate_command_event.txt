namespace = stargate_command_event

event = {
    id = stargate_command_event.1
    is_triggered_only = yes
	hide_window = yes

    immediate = {
        every_country = {
            limit = {
                is_ai = no
                is_regular_empire = yes
                has_country_flag = stargate_immersion_first_team_through
            }
            country_event = { id = stargate_command_event.2 }
        }
    }
}
country_event = {
    id = stargate_command_event.2
    is_triggered_only = yes
	hide_window = yes

    immediate = {
        random_list = {
            1 = {}
            99 = { country_event = { id = stargate_command_event.3 } }
        }
    }
}

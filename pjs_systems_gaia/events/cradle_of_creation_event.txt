namespace = cradle_of_creation_event

country_event = {
	id = cradle_of_creation_event.1

	hide_window = yes
	is_triggered_only = yes

	immediate = {
	    capital_scope = {
            solar_system = {
                every_system_planet = {
                    limit = { has_planet_flag = cradle_of_creation_cradle }

			        change_pc = "pc_gaia"

                    set_name = "Cradle"

                    while = { count = 1
                        add_district = district_city
                    }

                    while = { count = 2
                        add_district = district_generator
                    }

                    generate_betharian_deposit = yes
                    generate_gases_deposit = yes

                    if = {
                        limit = { owner = { is_regular_empire = yes } }

                        while = { count = 1
                            add_district = district_farming
                        }

                        add_building = building_factory_1
                        add_building = building_holo_theatres
                        add_building = building_commercial_zone
                    }

                    if = {
                        limit = { owner = { is_hive_empire = yes } }

                        while = { count = 1
                            add_district = district_farming
                        }

                        add_building = building_hive_warren
                        add_building = building_maintenance_depot
                        add_building = building_spawning_pool
                    }

                    if = {
                        limit = { owner = { is_machine_empire = yes } }

                        while = { count = 1
                            add_district = district_generator
                        }

                        add_building = building_drone_storage
                        add_building = building_maintenance_depot
                    }

                    while = {
                        count = 14
                        create_pop = { species = owner_main_species }
                    }
                }
            }
	    }
	}
}

                    create_colony = {
                        owner = root
                        species = root.owner_species
                        ethos = owner
                    }

                    planet_event = { id = game_start.21 }
                    planet_event = { id = game_start.22 }
                    planet_event = { id = game_start.23 }

                    set_name = "Father"

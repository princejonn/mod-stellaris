namespace = pjs_systems_gaia_event

planet_event = {
	id = pjs_systems_gaia_event.1

	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = { has_planet_flag = planet_cradle }

			change_pc = "pc_gaia"

			## gases
			add_deposit = d_bubbling_swamp

			## energy
			add_deposit = d_geothermal_vent

			## minerals
			add_deposit = d_rich_mountain

			## food
			add_deposit = d_tropical_island
		}
	}
}

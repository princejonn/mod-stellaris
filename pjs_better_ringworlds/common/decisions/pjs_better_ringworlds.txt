decision_rw_districts = {
	owned_planets_only = yes

	prerequisites = {
		"tech_rw_districts"
	}

	enactment_time = 720
	resources = {
		category = decisions
		cost = {
			influence = 100
			alloys = 2000
		}
	}

	potential = {
        is_planet_class = pc_ringworld_habitable
		owner = {
            has_ascension_perk = ap_master_builders
		}
		NOT = {
		    has_modifier = rw_districts
		}
	}

	effect = {
		add_modifier = {
			modifier = rw_districts
			days = -1
		}
	}

	ai_weight = {
		weight = 50
		modifier = {
			factor = 0
			free_district_slots > 0
		}
	}
}

decision_rw_spec_arms_industry = {
	owned_planets_only = yes

	prerequisites = {
		"tech_rw_spec_arms_industry"
	}

	enactment_time = 360
	resources = {
		category = decisions
		cost = {
			influence = 25
			alloys = 1000
		}
	}

	potential = {
        is_planet_class = pc_ringworld_habitable
		owner = {
            has_ascension_perk = ap_master_builders
		}
        AND = {
            NOT = { has_planet_flag = rw_spec_arms_industry }
            NOT = { has_planet_flag = rw_spec_bureaucracy }
            NOT = { has_planet_flag = rw_spec_civilian_industry }
            NOT = { has_planet_flag = rw_spec_culture }
            NOT = { has_planet_flag = rw_spec_science }
        }
	}

    effect = {
        set_planet_flag = rw_spec_arms_industry
	}

	ai_weight = {
		weight = 25
	}
}

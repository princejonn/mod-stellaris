	potential = {
		OR = {
			is_planet_class = pc_ringworld_habitable
			is_planet_class = pc_cybrex
		}
	}

	planet_modifier = {
        planet_district_rw_arms_industry_build_speed_mult = 0.25
		planet_building_foundry_1_build_speed_mult = 0.25
        planet_metallurgists_upkeep_mult = -0.2
	}

	weight_modifier = {
		base = 10
		modifier = {
			add = 5

		}
		modifier = {

			has_designation = col_ring_arms_industry
		}
		modifier = {
			add = 6
			num_districts = { type = district_rw_arms_industry value >= 2 }
		}
	}
}

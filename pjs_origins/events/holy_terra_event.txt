namespace = holy_terra_event

country_event = {
	id = holy_terra_event.1

	hide_window = yes
	is_triggered_only = yes

	immediate = {
	    capital_scope = {
            solar_system = {
                every_system_planet = {
                    limit = { has_planet_flag = holy_terra_earth }

                    set_name = "Holy Terra"
                    change_pc = "pc_city"

                    remove_all_districts = yes

                    while = { count = 5
                        add_district = district_arcology_housing
                    }
                    while = { count = 1
                        add_district = district_arcology_civilian_industry
                    }

                    add_building = building_precinct_house
                    add_building = building_research_lab_1
                    add_building = building_commercial_zone
                    add_building = building_holo_theatres

                    while = {
                        count = 20
                        create_pop = { species = owner_main_species }
                    }

                    set_graphical_culture = root

                    owner = {
                        add_resource = { energy = 500 }
                        add_resource = { minerals = 500 }
                        if = {
                            limit = { is_machine_empire = no }
                            add_resource = { food = 500 }
                        }
                        if = {
                            limit = { is_gestalt = no }
                            add_resource = { consumer_goods = 500 }
                        }

                        give_technology = { tech = "tech_mine_volatile_motes" message = no }
                        give_technology = { tech = "tech_mine_exotic_gases" message = no }
                        give_technology = { tech = "tech_mine_rare_crystals" message = no }
                    }
                }
            }
	    }
	}
}

country_event = {
	id = holy_terra_event.2

	hide_window = yes
	is_triggered_only = yes

	immediate = {
	    capital_scope = {
            solar_system = {
                every_system_planet = {
                    limit = { has_planet_flag = holy_terra_mars }

                    create_colony = {
                        owner = root
                        species = root.owner_species
                        ethos = owner
                    }

                    set_name = "Mars"
                    change_pc = "pc_continental"

			        clear_blockers = yes
			        clear_deposits = yes
                    remove_all_districts = yes

                    # 6 energy
                    generate_energy_deposits = yes

                    # 12 minerals
                    generate_mineral_deposits = yes
                    generate_mineral_deposits = yes

                    # 9 food
                    generate_food_deposits = yes
                    generate_food_1_deposit = yes
                    generate_food_2_deposit = yes

                    # 1 rare crystals
                    generate_crystals_deposit = yes

                    while = { count = 3
                        add_district = district_city
                    }
                    while = { count = 0
                        add_district = district_generator
                    }
                    while = { count = 7
                        add_district = district_mining
                    }
                    while = { count = 6
                        add_district = district_farming
                    }

                    remove_building = building_colony_shelter
                    add_building = building_capital
                    add_building = building_precinct_house
                    add_building = building_holo_theatres
                    add_building = building_holo_theatres
                    add_building = building_commercial_zone
                    add_building = building_resource_silo

                    add_building = building_crystal_mines

                    while = { count = 44
                        create_pop = {
                            species = owner
                        }
                    }
                }
            }
	    }
	}
}
